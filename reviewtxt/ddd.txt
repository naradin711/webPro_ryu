create table store (
            storenum        number(8) PRIMARY KEY,
            storeimage      varchar2(2000) not null,
            storename       varchar2(100) not null,
            storeaddress    varchar2(2000) not null,
            storetel        varchar2(100) not null,
            storetype       varchar2(100) not null,
            storeprice      varchar2(100) not null,
            storetime       varchar2(100) not null,
            storeholiday    varchar2(100) not null,
            storemenu1      varchar2(100) not null,
            storemenu1cost  varchar2(100) not null,
            storemenu2      varchar2(100),
            storemenu2cost  varchar2(100),
            storemenu3      varchar2(100),
            storemenu3cost  varchar2(100),
            storemenu4      varchar2(100),
            storemenu4cost  varchar2(100),
            storemenu5      varchar2(100),

            searchtag       varchar2(100),
            storeconfirm    char(1) default 'N',

            storescore      number(8) default 0
    );

drop table storereview;
drop SEQUENCE storereview_sq;
create SEQUENCE storereview_sq MAXVALUE 999999 NOCACHE NOCYCLE;
create table storereview (
            reviewnum         number(8) PRIMARY KEY,
            storenum          number(8) REFERENCES store(storenum),
            userprofile       varchar2(1000) not null,
            userid            varchar2(100),
            reviewcontent     varchar2(2000) not null,
            reviewimage1      varchar2(500),
            reviewimage2      varchar2(500),
            reviewimage3      varchar2(500),
            reviewimage4      varchar2(500),
            reviewimage5      varchar2(500),
            reviewscore       number(3) default 0,
            reviewdate        date default sysdate
    );
